CREATE TABLE users (
	user_id int(11) NOT NULL AUTO_INCREMENT,
	name varchar(50) NOT NULL,
	email varchar(50) NOT NULL,
	password varchar(100) NOT NULL,
	registered datetime NOT NULL,

	PRIMARY KEY(user_id)
);

CREATE TABLE bouquets (
	bouquet_id int(11) NOT NULL AUTO_INCREMENT,
	name varchar(50),
	description varchar(500),
	pack_size int(11),
	image varchar(50),
	collections varchar(500),
	tags varchar(500),
	date_added datetime NOT NULL,

	PRIMARY KEY(bouquet_id)
);

CREATE TABLE srps (
	srp_id int(11) NOT NULL AUTO_INCREMENT,
	bouquet_id int(11) NOT NULL,
	image varchar(50),
	name varchar(50),
	srp decimal(10,2),
	stems int(11),
	date_added datetime NOT NULL,

	PRIMARY KEY(srp_id),
	FOREIGN KEY(bouquet_id) REFERENCES bouquets(bouquet_id)
);

CREATE TABLE collections (
	collection_id int(11) NOT NULL AUTO_INCREMENT,
	name varchar(50),
	description varchar(500),
	banner_image varchar(50) NOT NULL,
	date_added datetime NOT NULL,

	PRIMARY KEY(collection_id)
);

CREATE TABLE collection_item (
	collection_item_id int(11) NOT NULL AUTO_INCREMENT,
	collection_id int(11) NOT NULL,
	bouquet_id int(11) NOT NULL,
	srp_id int(11),

	PRIMARY KEY(collection_item_id),
	FOREIGN KEY(bouquet_id) REFERENCES bouquets(bouquet_id),
	FOREIGN KEY(collection_id) REFERENCES collections(collection_id),
	FOREIGN KEY(srp_id) REFERENCES srps(srp_id)
)
